from interfaces import AggregatorV3Interface
import get_price_module

MINIMUM_USD: public(constant(uint256)) = as_wei_value(5, "ether")
PRICE_FEED: public(immutable(AggregatorV3Interface))
OWNER: public(immutable(address))

PRECISION: constant(uint256) = 1 * (10 ** 18)

funders: public(DynArray[address, 1000])
funders_to_amount_funded: public(HashMap[address, uint256])

@deploy
def __init__(price_feed: address):
    PRICE_FEED = AggregatorV3Interface(price_feed)
    OWNER = msg.sender

@external
@payable
def fund():
# allows users to send $ to this contract with a minimum amount
    usd_value_of_eth: uint256 = get_price_module._get_eth_to_usd_rate(PRICE_FEED, msg.value)
    assert usd_value_of_eth >= MINIMUM_USD, "You must spend more ETH!"
    self.funders.append(msg.sender)
    self.funders_to_amount_funded[msg.sender] += msg.value

@external 
def withdraw():
    assert msg.sender == OWNER, "Not the contract owner!"
    raw_call(OWNER, b"", value = self.balance)
    # resetting the funders dynamic array & the funders_to_amount_funded HashMap
    for funder: address in self.funders:
        self.funders_to_amount_funded[funder] = 0
    self.funders = []

# view functions
@external
@view
def get_version() -> uint256:
    return staticcall PRICE_FEED.version()

@external
@view
def get_funder(index: uint256) -> address:
    return self.funders[index]

@external
@view
def get_owner() -> address:
    return OWNER

@external
@view
def get_eth_to_usd_rate(eth_amount: uint256) -> uint256:
    return get_price_module._get_eth_to_usd_rate(PRICE_FEED, eth_amount)

#default function
@external
@payable
def __default__():
    pass
